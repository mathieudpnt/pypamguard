 61        """
 62        Read data from the file. This function is polymorphic in the sense that it
 63        can be used for any of the following purposes:
 64
 65        1. Read in a single value of a given datatype (for example `read_numeric(DTYPES.INT32)`).
 66        2. Read in an array of values of a given datatype (for example `read_numeric(DTYPES.INT32, (5,))`).
 67        3. Read in an n-dimensional array of values of a given datatype (for example `read_numeric(DTYPES.INT32, (5, 5))`).
 68        4. Read in an interleaved array of values of a given datatype (for example `read_numeric([DTYPES.INT32, DTYPES.INT32], (5,))`).
 69
 70        Read in an array of 5 integers (32-bit).
 71        Return a single `np.ndarray` of type `np.int32`.
 72        ```python
 73        bin_read(DTYPES.INT32, (5,))
 74        ```
 75
 76        Read in two 5-length arrays (interleaved int32, int32).
 77        Return a tuple of two `np.ndarray`s of type `np.int16`
 78        and `np.int64`.
 79        ```python
 80        bin_read([DTYPES.INT16, DTYPES.INT64], (5,))
 81        ```
 82
 83        Read in a single float (32-bit) and divide by 100.
 84        Return a single `np.float32`.
 85        ```python
 86        bin_read((DTYPES.FLOAT32, lambda x: x / 100))
 87        ```
 88
 89        Read in two 5-length arrays (interleaved float32, int8).
 90        Divide the int8 array by 100.
 91        Return a tuple of two `np.ndarray`s of type `np.float32` and `np.float32`.
 92        (NOTE: the int8 array is returned as a float32 array due to the division by 100.)
 93        ```python
 94        bin_read([(DTYPES.FLOAT32), (DTYPES.INT8, lambda x: x/100)], (5,))
 95        ```
 96
 97        Read in a single 5x2 array of floats (32-bit).
 98        Return a 2d `np.ndarray` of type `np.float32`.
 99        ```python
100        bin_read(DTYPES.FLOAT32, (5, 2))
101        ```
102        """
103        if type(shape) != tuple: shape = (shape,)
104        dtypes = [(dtype_i, None) if isinstance(dtype_i, DTYPES) else dtype_i for dtype_i in ([dtype] if not isinstance(dtype, list) else dtype)]
105        data = np.frombuffer(self.__read(sum(dtype_i[0].value.itemsize for dtype_i in dtypes) * np.prod(shape)   ), dtype=[(f'f{i}', dtype_i[0].value.newbyteorder(self._endianess)) for i, dtype_i in enumerate(dtypes)])
106        ret_val = tuple(self.__collate(data, dtypes, shape))
107        return ret_val[0] if len(ret_val) == 1 else ret_val
